country_event = {
	id = WSG_event.001
	title = WSG_event.001.t
	desc = WSG_event.001.d
	is_triggered_only = yes
	fire_only_once = yes
	option = {
		name = WSG_event.001.a
		add_offsite_building = { type = industrial_complex level = 100 }
		add_offsite_building = { type = arms_factory level = 15 }
		add_offsite_building = { type = dockyard level = 25 }
		set_variable = { WSG_panci = 300 }
		custom_effect_tooltip =  WSG_panci_set_300_tt
		hidden_effect = {
			set_variable = { WSG_monthly_panci = 5 }
		}
	}
	option = {
		name = WSG_event.001.b
		set_power_balance = {
	        id = WSG_balance_of_power
	        set_value = 0
        }
		set_variable = { WSG_panci = -300 }
		custom_effect_tooltip  = WSG_panci_set_f300_tt
		hidden_effect = {
			set_variable = { WSG_monthly_panci = 30 }
		}
	}
	option = {
		name = WSG_event.001.c
		set_power_balance = {
	        id = WSG_balance_of_power
	        set_value = 1
        }
		set_variable = { WSG_panci = -1500 }
		custom_effect_tooltip  = WSG_panci_set_f1500_tt
		hidden_effect = {
			set_variable = { WSG_monthly_panci = 50 }
		}
		every_state = {
			limit = {
				is_controlled_by = WSG
			    is_core_of = WSG
				industrial_complex > 3
			}
			remove_building = {
            type = industrial_complex
            level = 4
            }
		}
	}
}

		has_completed_focus = WSG_secretary_lexington
