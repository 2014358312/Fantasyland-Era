country_event = {
	id = WSG_event.001
	title = WSG_event.001.t
	desc = WSG_event.001.d
	is_triggered_only = yes
	fire_only_once = yes
	option = {
		name = WSG_event.001.a
		add_offsite_building = { type = industrial_complex level = 100 }
		add_offsite_building = { type = arms_factory level = 15 }
		add_offsite_building = { type = dockyard level = 25 }
		set_variable = { WSG_panci = 300 }
		custom_effect_tooltip =  WSG_panci_set_300_tt
	}
	option = {
		name = WSG_event.001.b
		set_power_balance = {
	        id = WSG_balance_of_power
	        set_value = 0
        }
		set_variable = { WSG_panci = -300 }
		custom_effect_tooltip  = WSG_panci_set_f300_tt
	}
	option = {
		name = WSG_event.001.c
		set_power_balance = {
	        id = WSG_balance_of_power
	        set_value = 1
        }
		set_variable = { WSG_panci = -1500 }
		custom_effect_tooltip  = WSG_panci_set_f1500_tt
		every_state = {
			limit = {
				is_controlled_by = WSG
			    is_core_of = WSG
				industrial_complex > 3
			}
			remove_building = {
            type = industrial_complex
            level = 4
                }
			}
	}
}

country_event = {
	id = WSG_inner_circle_lexington.01

	hidden = yes

	is_triggered_only = yes

	trigger = {
		has_completed_focus = WSG_secretary_lexington
		OR = {
			check_variable = { WSG_ascended_advisor_1 = 1 }
			check_variable = { WSG_ascended_advisor_2 = 1 }
			check_variable = { WSG_ascended_advisor_3 = 1 }
		}
	}

	immediate = {
		if = {
			limit = {
				NOT = { has_completed_focus = WSG_lexington_plan }

			country_event = { id = WSG_inner_circle_lexington.02 days = 70 }
		}
		else_if = {
			limit = {
				NOT = { has_completed_focus = WSG_expand_resource_output }
			}
			activate_shine_on_focus = WSG_expand_resource_output

		}

			limit = {
				NOT = { has_completed_focus = WSG_ship_girl_reserve_army }
			}
			activate_shine_on_focus = WSG_ship_girl_reserve_army

		}
	}
