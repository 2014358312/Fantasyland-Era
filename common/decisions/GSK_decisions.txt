		}
		available = {
			custom_trigger_tooltip = {
				tooltip = GSK_Restore_the_ecosystem_tt

			}

				}
			}

	}

		cost = 120
		days_remove = 60
		visible = {

		}
		available = {
			custom_trigger_tooltip = {
				tooltip = GSK_Expand_the_central_area_tt
				always = yes
			}
		}
		modifier = {
			conscription = -0.05
			consumer_goods_expected_value = 0.1
			production_speed_buildings_factor = 0.1
		}
		remove_effect = {
			effect_tooltip = {
				swap_ideas = {
					remove_idea = GSK_rebuild_Gensokyo_idea
					add_idea = GSK_rebuild_Gensokyo_idea_2
				}
