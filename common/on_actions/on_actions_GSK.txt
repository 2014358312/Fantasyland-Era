	on_startup = {
		effect = {
			GSK = {
				#she now is in state of division on start.

					add_idea = GSK_hakurei_great_barrier_protection
				}
			}
			HBG = {
				every_core_state = {
					add_building_construction = {
						type = bunker
						level = 10
						instant_build = yes
						province = {
							all_provinces = yes
						}
					}
				}
			}
		}
	}

		effect = {
			set_variable = {

				GSK_Planning_idea_3_war_core = num_owned_controlled_states
			}

				GSK_Planning_idea_3_war_core = 0.05
			}

				GSK_Planning_idea_3_war_core = 0.1
			}
		}
	}
	on_monthly_GSK = {
		effect = {
			if = {
				limit = {
					has_idea = GSK_Case_of_Princess_Kaguya_idea
				}
				capital_scope = {
					create_unit = {
					    division = "name = \"Kaguya's Tank Division\" division_template = \"Kaguya's Tank Division\" start_experience_factor = 0.5"

    					count = 1
					}
				}
			}
