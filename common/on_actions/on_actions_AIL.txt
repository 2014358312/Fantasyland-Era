on_actions = {
	on_startup = {
		effect = {
			AIL = {
				if = {
					limit = { has_country_flag = AIL_neuro_flag }
					AIL_Neuro_sama = {
						add_country_leader_role = {
							country_leader = {
								ideology = hive_mind_n_subtype
								traits = { neuro_sama_blue }
								desc = "POLITICS_NEURO_SAMA_DESC"
								expire = "2200.1.1.1"
								id = -1
							}
							promote_leader = yes
						}
					}
					set_politics = {
						ruling_party = neutrality
						elections_allowed = no
					}
					set_popularities = {
						neutrality = 100
					}
				}
			}
		}
	}
	
	on_monthly_AIL = {
		effect = {
			if = {
				limit = {
					is_ai = yes
					has_country_flag = eylstadt_drift_defence_war_member_flag
				}
				set_fuel_ratio = 1.00
				if = {
					limit = { has_equipment = { infantry_equipment_1 < 1 } }
					add_equipment_to_stockpile = {
						type = infantry_equipment_1
						amount = 10000
					}
				}
				if = {
					limit = { has_equipment = { support_equipment_1 < 1 } }
					add_equipment_to_stockpile = {
						type = support_equipment_1
						amount = 2000
					}
				}
				if = {
					limit = { has_equipment = { artillery_equipment_1 < 1 } }
					add_equipment_to_stockpile = {
						type = artillery_equipment_1
						amount = 1000
					}
				}
				if = {
					limit = { has_equipment = { motorized_equipment_1 < 1 } }
					add_equipment_to_stockpile = {
						type = motorized_equipment_1
						amount = 800
					}
				}
			}
		}
	}
	
	on_war = {
		effect = {
			every_country = {
				limit = {
					has_character = AIL_Neuro_sama
					has_government = neutrality
					has_country_leader = {
						character = AIL_Neuro_sama
						ruling_only = yes
					}
					has_country_leader_ideology = hive_mind_n_subtype
					has_country_flag = AIL_neuro_flag
				}
				AIL_Neuro_sama = {
					remove_country_leader_role = {
						ideology = hive_mind_n_subtype
					}
					add_country_leader_role = {
						country_leader = {
							ideology = hive_mind_f_subtype
							traits = { neuro_sama_red }
							desc = "POLITICS_NEURO_SAMA_DESC"
							expire = "2200.1.1.1"
							id = -1
						}
						promote_leader = yes
					}
				}
				set_politics = {
					ruling_party = fascism
					elections_allowed = no
				}
				set_popularities = {
					fascism = 100
				}
			}
		}
	}
	
	on_peace = {
		effect = {
			every_country = {
				limit = {
					has_character = AIL_Neuro_sama
					has_government = fascism
					has_country_leader = {
						character = AIL_Neuro_sama
						ruling_only = yes
					}
					has_country_leader_ideology = hive_mind_f_subtype
					has_country_flag = AIL_neuro_flag
				}
				AIL_Neuro_sama = {
					remove_country_leader_role = {
						ideology = hive_mind_f_subtype
					}
					add_country_leader_role = {
						country_leader = {
							ideology = hive_mind_n_subtype
							traits = { neuro_sama_blue }
							desc = "POLITICS_NEURO_SAMA_DESC"
							expire = "2200.1.1.1"
							id = -1
						}
						promote_leader = yes
					}
				}
				set_politics = {
					ruling_party = neutrality
					elections_allowed = no
				}
				set_popularities = {
					neutrality = 100
				}
			}
		}
	}
}
