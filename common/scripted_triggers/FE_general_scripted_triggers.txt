has_military_academy = {
	if = {
		limit = { 
			NOT = { has_country_flag = ROOT_has_military_academy_flag }
		}
		custom_trigger_tooltip = {
			tooltip = has_military_academy_tt 
			has_country_flag = ROOT_has_military_academy_flag
		}
	}
}
