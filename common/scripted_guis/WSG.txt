		window_name = "WSG_panci"
		context_type = decision_category

		triggers = {
			open_dininghall_btn_click_enabled = {
				has_completed_focus = WSG_offspring_of_the_ancients
			}
		}
		effects = {
			open_dininghall_btn_click = {
				
				if = {
					limit = {
						has_country_flag = show_dininghall_window
					}
					clr_country_flag = show_dininghall_window
					
				}
				else = {
					set_country_flag = show_dininghall_window
					
				}
			}
			
		}

					limit = {

						feiyuguantou_cishu = 1
					}
					
				}

					limit = {
						check_variable = {
							feiyuguantou_cishu > 0
						}

				}
			random_list = {
				10 = {  country_event = { days = 0 id = feiyuguantou_shijian.5 } }
                
				30 = {  country_event = { days = 0 id = feiyuguantou_shijian.6 } }
				
				
				}

				var = WSG_panci
				value = -25
			}

			custom_effect_tooltip = menu_1_btn_tt					
	   	}

				value = -50
			}

            idea = WSG_tiandouhua_buff
            days = 60
            }
			set_country_flag = { flag = WSG_start_cooking value = 1 days = 60 } 
			custom_effect_tooltip = menu_2_btn_tt
		}
