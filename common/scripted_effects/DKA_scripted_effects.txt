DKA_improve_government_efficiency = {
	custom_effect_tooltip = DKA_improve_government_efficiency_tt
	if = {
		limit = {
			has_idea = DKA_government_efficiency_1
		}
		swap_ideas = {
			remove_idea = DKA_government_efficiency_1
			add_idea = DKA_government_efficiency_2
		}
	}
	else_if = {
		limit = {
			has_idea = DKA_government_efficiency_2
		}
		swap_ideas = {
			remove_idea = DKA_government_efficiency_2
			add_idea = DKA_government_efficiency_3
		}
	}
	else_if = {
		limit = {
			has_idea = DKA_government_efficiency_3
		}
		swap_ideas = {
			remove_idea = DKA_government_efficiency_3
			add_idea = DKA_government_efficiency_4
		}
	}
	else_if = {
		limit = {
			has_idea = DKA_government_efficiency_4
		}
		swap_ideas = {
			remove_idea = DKA_government_efficiency_4
			add_idea = DKA_government_efficiency_5
		}
	}
	else = {
		effect_tooltip = {
			swap_ideas = {
				remove_idea = DKA_government_efficiency_1
				add_idea = DKA_government_efficiency_2
			}
		}
	}
}

DKA_improve_brain = {
	meta_effect = {
		text = {
			if = {
				limit = {
					has_idea = DKA_believe_in_brute_force_[INQUIPMENT]_0
				}
				swap_ideas = {
					remove_idea = DKA_believe_in_brute_force_[INQUIPMENT]_0
					add_idea = DKA_believe_in_brute_force_[INQUIPMENT]_1
				}
			}
			else_if = {
				limit = {
					has_idea = DKA_believe_in_brute_force_[INQUIPMENT]_1
				}
				swap_ideas = {
					remove_idea = DKA_believe_in_brute_force_[INQUIPMENT]_1
					add_idea = DKA_believe_in_brute_force_[INQUIPMENT]_2
				}
			}
			else_if = {
				limit = {
					has_idea = DKA_believe_in_brute_force_[INQUIPMENT]_2
				}
				swap_ideas = {
					remove_idea = DKA_believe_in_brute_force_[INQUIPMENT]_2
					add_idea = DKA_believe_in_brute_force_[INQUIPMENT]_3
				}
			}
			else_if = {
				limit = {
					has_idea = DKA_believe_in_brute_force_[INQUIPMENT]_3
				}
				if = {
					limit ={
						check_variable = { DKA_believe_in_brute_force_inquipment_step_var = 2 }
					}
					remove_ideas = DKA_believe_in_brute_force_[INQUIPMENT]_3
				}
				else = {
					swap_ideas = {
						remove_idea = DKA_believe_in_brute_force_[INQUIPMENT]_3
						add_idea = DKA_believe_in_brute_force_[INQUIPMENT]_4
					}
				}
			}
			else = {
				effect_tooltip = {
					swap_ideas = {
						remove_idea = DKA_believe_in_brute_force_0_0
						add_idea = DKA_believe_in_brute_force_0_1
					}
				}
			}
		}
		INQUIPMENT = [?DKA_believe_in_brute_force_inquipment_step_var]
	}
}
