LAN_reset_degree_of_colonization = {
	every_country = {
		clear_variable = ROOT_influence_from_landsol_var
		if = {
			limit = { has_idea = LAN_landsol_influence_economy }
			add_to_variable = { ROOT_influence_from_landsol_var = 1 }
		}
		if = {
			limit = { has_idea = LAN_landsol_influence_military }
			add_to_variable = { ROOT_influence_from_landsol_var = 1 }
		}
		if = {
			limit = { has_idea = LAN_landsol_influence_political }
			add_to_variable = { ROOT_influence_from_landsol_var = 1 }
		}
	}
}

		subtract_from_variable = { party_troop_seats = party_union_seats }

LAN_round_inner_circle_vertex_pos = {
	set_variable = { LAN_inner_circle_coordinate_pos = 0 }
	set_variable = { LAN_near_pos_min_sq = 100 }
	for_each_loop = {
		array = LAN_inner_circle_coordinate_x_array
		index = i
		value = v
		
		set_variable = { LAN_near_pos_x_abs = LAN_inner_circle_coordinate_x_array^i }
		subtract_from_variable = { LAN_near_pos_x_abs = LAN_inner_circle_vertex_x }
		set_variable = { LAN_near_pos_x_sq = LAN_near_pos_x_abs }
		multiply_variable = { LAN_near_pos_x_sq = LAN_near_pos_x_abs }
		set_variable = { LAN_near_pos_y_abs = LAN_inner_circle_coordinate_y_array^i }
		subtract_from_variable = { LAN_near_pos_y_abs = LAN_inner_circle_vertex_y }
		set_variable = { LAN_near_pos_y_sq = LAN_near_pos_y_abs }
		multiply_variable = { LAN_near_pos_y_sq = LAN_near_pos_y_abs }
		if = {
			limit = {
				check_variable = {
					var = LAN_near_pos_x_sq
					value = LAN_near_pos_min_sq
					compare = less_than
				}
				check_variable = {
					var = LAN_near_pos_y_sq
					value = LAN_near_pos_min_sq
					compare = less_than
				}
			}
			set_variable = { LAN_near_pos_sq = LAN_near_pos_x_sq }
			add_to_variable = { LAN_near_pos_sq = LAN_near_pos_y_sq }
			if = {
				limit = { check_variable = { LAN_near_pos_sq < LAN_near_pos_min_sq } }
				set_variable = { LAN_near_pos_min_sq = LAN_near_pos_sq }
				set_variable = { LAN_inner_circle_coordinate_pos = i }
				log = "[GetDateText]: [Root.GetName]: new coordinate x is [?LAN_inner_circle_coordinate_x_array^i]."
				log = "[GetDateText]: [Root.GetName]: new coordinate y is [?LAN_inner_circle_coordinate_y_array^i]."
				add_to_variable = { LAN_inner_circle_coordinate_pos = 1 }
			}
		}
	}
	log = "[GetDateText]: [Root.GetName]: var of LAN_inner_circle_coordinate_pos is [?LAN_inner_circle_coordinate_pos] now."
	clear_variable = LAN_near_pos_x_abs
	clear_variable = LAN_near_pos_x_sq
	clear_variable = LAN_near_pos_y_abs
	clear_variable = LAN_near_pos_y_sq
	clear_variable = LAN_near_pos_min_sq
}
